Create flat geometric forms with Three.js.

Flat Angular Shape

// Create flat angular polygon function createFlatShape(vertices, color) { const shape = new THREE.Shape(); for (let i = 0; i < vertices; i++) { const angle = (Math.PI * 2 / vertices) * i; const radius = 2 + Math.random() * 0.5; const x = Math.cos(angle) * radius; const y = Math.sin(angle) * radius; if (i === 0) { shape.moveTo(x, y); } else { shape.lineTo(x, y); // Sharp lines, no curves } } shape.closePath(); const geometry = new THREE.ShapeGeometry(shape); const material = new THREE.MeshBasicMaterial({ color: color, side: THREE.DoubleSide }); return new THREE.Mesh(geometry, material); }
