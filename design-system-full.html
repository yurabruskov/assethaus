Code Examples

Create flat geometric forms with Three.js.

Flat Angular Shape with Three.js

// Create flat angular polygon with sharp edges function createFlatShape(vertices, color) { const shape = new THREE.Shape(); for (let i = 0; i < vertices; i++) { const angle = (Math.PI * 2 / vertices) * i; const radius = 1.5 + Math.random() * 0.8; const x = Math.cos(angle) * radius; const y = Math.sin(angle) * radius; if (i === 0) { shape.moveTo(x, y); } else { shape.lineTo(x, y); // Sharp lines, NO curves! } } shape.closePath(); // Use ShapeGeometry for flat 2D shape in 3D space const geometry = new THREE.ShapeGeometry(shape); // MeshStandardMaterial with emissive glow const material = new THREE.MeshStandardMaterial({ color: color, metalness: 0.5, roughness: 0.3, side: THREE.DoubleSide, emissive: color, emissiveIntensity: 0.2 }); const mesh = new THREE.Mesh(geometry, material); // Animate rotation function animate() { requestAnimationFrame(animate); mesh.rotation.y += 0.01; mesh.rotation.x += 0.005; } animate(); return mesh; }
